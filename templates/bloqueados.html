@app.route("/bloqueados") def bloqueados(): if not login_required(): return redirect("/login") conn = get_conn() cur = conn.cursor() cur.execute(""" SELECT id, ip, motivo, data_cadastro FROM ips_bloqueados ORDER BY data_cadastro DESC """) rows = cur.fetchall() cur.close() conn.close() bloqueados = [ { "id": r[0], "ip": r[1], "motivo": r[2], "data": str(r[3]) if r[3] else "" } for r in rows ] return render_template( "bloqueados.html", bloqueados=bloqueados, usuario=session.get("nome") )
